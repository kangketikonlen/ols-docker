setuidmode 2
vhroot /var/www/vhosts/$VH_NAME/
configfile $SERVER_ROOT/conf/vhosts/$VH_NAME/vhconf.conf
allowsymbollink 1
restrained 1
enablescript 1

virtualhostconfig  {
    enablegzip 1
    docroot $VH_ROOT/html/

    accesslog $SERVER_ROOT/logs/$VH_NAME.access.log {
        useserver 0
        rollingsize 100M
        keepdays 7
        compressarchive 1
    }

    scripthandler  {
        add lsapi:lsphp lsphp74
    }

    rewrite  {
        enable 1
        autoloadhtaccess 1
        loglevel 0
    }

    errorlog  {
        useserver 1
    }

    expires  {
        enableexpires 1
    }

    accesscontrol  {
        allow *
    }

    index  {
        autoindexuri /_autoindex/default.php
        autoindex 0
        indexfiles index.html, index.php
        useserver 0
    }
}
